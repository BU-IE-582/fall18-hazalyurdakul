For euclidean distance, I tried 20 k values to find its optimal value. Misclassification error which is the mean of misclassified instances is calculated and plotted for each k.

set.seed(123)
y = X_TRAIN$V1
idx <- createFolds(y, k = 10)  #creating 10 folds 
sapply(idx, length)  #shows the number of instants of each fold
## Fold01 Fold02 Fold03 Fold04 Fold05 Fold06 Fold07 Fold08 Fold09 Fold10 
##     89     90     90     90     89     89     89     90     89     91
start.time <- Sys.time()
k.trials <- 1:20
res <- sapply(k.trials, function(k) {
    ## try out each version of k from 1 to 20
    res.k <- sapply(seq_along(idx), function(i) {
        ## loop over each of the 10 cross-validation folds predict the
        ## held-out samples using k nearest neighbors
        pred = kknn(as.factor(y[-idx[[i]]]) ~ ., as.data.frame(traindata[-idx[[i]], 
            ]), as.data.frame(traindata[idx[[i]], ]), k = k, 
            distance = 2)
        ## the ratio of misclassified samples
        mean(as.factor(y[idx[[i]]]) != pred$fit)
    })
    ## average over the 10 folds
    mean(res.k)
})
end.time <- Sys.time()
time.taken <- end.time - start.time

plot(k.trials, res, type = "o", ylab = "misclassification error", 
    xlab = "k trial", main = "Optimizing k value wrt misclassification error")

which(res == min(res))
